    USERS {
        INT user_id PK
        VARCHAR username
        VARCHAR email
        VARCHAR password_hash
        DATETIME registration_date
        VARCHAR last_visit_ip
        VARCHAR role "NULL | USERS | CSR | ADMIN | DEVELOPER"
        VARCHAR status "ONLINE | OFFLINE"
        VARCHAR permission "ACTIVE | DEACTIVE | DELETED"
    }

    GAMES {
        INT game_id PK
        INT developer_id FK
        VARCHAR title
        TEXT description
        DECIMAL price
        DECIMAL overall_rating
        VARCHAR status "DRAFT | ACTIVE | TAKEN_DOWN"
    }

    TAGS {
        INT tag_id PK
        VARCHAR tag_name
    }

    GAME_TAGS {
        INT game_id PK, FK
        INT tag_id PK, FK
    }

    GAME_MEDIA {
        INT media_id PK
        INT game_id FK
        VARCHAR file_url
        VARCHAR media_type "media | game_file"
    }

    TRANSACTIONS {
        INT transaction_id PK
        INT user_id FK
        DECIMAL total_amount
        DATETIME transaction_date
        VARCHAR receipt_number
    }

    TRANSACTION_ITEMS {
        INT item_id PK
        INT transaction_id FK
        INT game_id FK
        DECIMAL purchase_price
    }

    SHOPPING_CARTS {
        INT cart_id PK
        INT user_id FK
        INT game_id FK
        DATETIME added_at
    }

    GAME_LICENSES {
        INT license_id PK
        INT game_id FK
        INT user_id FK
        INT transaction_item_id FK
        DATETIME acquired_date
        VARCHAR status "ACTIVE | REVOKED | FROZEN"
    }

    WISH_LISTS {
        INT wishlist_id PK
        INT user_id FK
        INT game_id FK
        DATETIME added_at
    }

    REFUND_REQUESTS {
        INT refund_id PK
        INT buyer_id FK
        INT transaction_item_id FK
        INT handled_by FK "nullable"
        TEXT reason
        TEXT reject_reason
        DATETIME created_at
        DATETIME resolved_at
        VARCHAR status "PENDING | APPROVED | REJECTED"
    }

    REVIEWS {
        INT review_id PK
        INT game_id FK
        INT user_id FK
        TEXT content
        DATETIME created_at
        VARCHAR attitude "POSITIVE | NEGATIVE"
        VARCHAR status "VISIBLE | HIDDEN | DELETED"
    }

    REVIEW_REPLYS {
        INT review_reply_id PK
        INT review_id FK
        INT user_id FK
        INT parent_reply_id FK "nullable"
        TEXT content
        DATETIME created_at
        VARCHAR status "VISIBLE | HIDDEN | DELETED"
    }

    FRIENDSHIPS {
        INT friendship_id PK
        INT sender_id FK
        INT receiver_id FK
        DATETIME created_at
        VARCHAR status "PENDING | ACCEPTED | DECLINED"
    }

    MESSAGES {
        INT message_id PK
        INT sender_id FK
        INT receiver_id FK
        TEXT content
        DATETIME sent_at
        BOOLEAN is_read
    }

    BLACKLISTS {
        INT blacklist_id PK
        INT blocker_id FK
        INT blocked_id FK
        DATETIME created_at
    }

    ILEARN_PINGS {
        INT id PK
        DATETIME checked_at
        INT latency_ms
        VARCHAR status "UP | DOWN"
    }

    ILEARN_REPORTS {
        INT id PK
        DATETIME reported_at
    }
